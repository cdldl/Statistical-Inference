import numpy.random as npr
M = 20000
L = 1000
alpha = 0.1
n_BC = np.zeros(L)
n_BHP = np.zeros(L)
FDR_BHP = np.zeros(L)
indP = np.arange(M) + 1

for i in range(L):
    # Simulate P-values
    y = npr.binomial(1, 0.1, M)
    P = np.zeros(M)
    P[y==0] = npr.uniform(0, 1, np.sum(y==0))
    P[y==1] = npr.beta(0.5, 1, np.sum(y==1))
    
    # Bonferroni correction
    n_BC[i] = np.sum(P < (alpha/M))
    
    # Benjamini-Hochberg Procedure
    ind_sort = np.argsort(P)
    P_sort = P[ind_sort]
    y_sort = y[ind_sort]
    Lset = np.where((P_sort < (indP/M*alpha)) == True)[0]
    if (len(Lset) > 0):
        bh = Lset[-1] + 1
        n_BHP[i] = bh
        FDR_BHP[i] = np.sum(y_sort[:bh] == 0) / bh

Problem 3¶

Observations are $z_i$, latent variables are $\mu_i$, parameter is $\alpha$. The complete-data likelihood is written as $$p(z,\mu|\alpha)=\prod_{i=1}^N\mathcal{N}(z_i|\mu_i,1)\mathcal{N}(\mu_i|0,\alpha^{-1}).$$ This implies the posterior distribution (E-step): $$p(\mu_i|z_i,\alpha)=\mathcal{N}(\nu_i,\sigma^2),$$ $$\sigma^2=\frac{1}{1+\alpha},\,\nu_i=z_i\sigma^2.$$ Then $Q$ function is obtained as $$Q(\alpha)=\mathbb{E}_{z|x,\alpha^{\text{old}}}\,(\log p(z,\mu|\alpha))=-N\log(2\pi)+\frac{N}{2}\log(\alpha)+\sum_{i=1}^N[-\frac{1+\alpha}{2}(\nu_i^2+\sigma^2)+z_i\nu_i-\frac12 z_i^2].$$ So the M-step updating equation is $$\alpha=\frac{N}{\sum_{i=1}^N(\nu_i^2+\sigma^2)}.$$ After EM algorithm converges, we obtain the estimate of $\mu$ by $$\hat{\mu_i}=\frac{z_i}{1+\hat{\alpha}}.$$

Note that EM is doing MLE while JSE uses the method of moments.

N = 5
alpha = 1
n_rep = 50
n_iter = 5000
tol = 1e-6
logls = np.zeros(n_iter+1)
err_em = np.zeros(n_rep)
err_js = np.zeros(n_rep)
for i in range(n_rep):
    # Simulate data
    mu = npr.normal(0, np.sqrt(1/alpha), N)
    z = mu + npr.normal(0, 1, N)
    
    # EM algorithm
    # initialize
    alpha_hat = 2
    logls[0] = - N/2*np.log(2*np.pi*(1+1/alpha_hat)) - 1/2/(1+1/alpha_hat) * np.sum(z**2) - 1e20
    # iterations
    for iter in range(n_iter):
        # E-step
        sigma_sq = 1 / (1 + alpha_hat)
        nu = z * sigma_sq
        # check log-likelihood
        logl = - N/2*np.log(2*np.pi*(1+1/alpha_hat)) - 1/2/(1+1/alpha_hat) * np.sum(z**2)
        logls[iter + 1] = logl
        imp = logls[iter + 1] - logls[iter]
        if (imp < 0):
            print("log-likelihood decreases...")
            break
        if (np.abs(imp / logls[iter]) < tol):
#             print("EM algorithm converges in %s iterations." % (iter+1))
            break
        # M-step
        alpha_hat = N / np.sum(nu**2 + sigma_sq)
    mu_hat_em = z / (1 + alpha_hat)
    err_em[i] = np.mean((mu_hat_em - mu) ** 2)
        
    # James-Stein estimator
    mu_hat_js = (1 - (N-2)/np.sum(z**2)) * z
    err_js[i] = np.mean((mu_hat_js - mu) ** 2)

Problem 4¶

(a)¶

Based on the fact that $$p(\mu_i|y_i,\alpha)\propto p(\mu_i|\alpha)p(y_i|\mu_i),$$ we obtain the updating equation for E-step as $$p(\mu_i|y_i,\alpha)=\text{Dir}(\mu_i|\nu_i),\,\nu_{ik}=\alpha_k+y_{ik}.$$ Note that $$\mathbb{E}_{\mu_i|y_i}[\log(\mu_{ik})]=\psi(\nu_{ik})-\psi(\sum_{j=1}^K\nu_{ij}).$$ Then we obtain the $Q$ function as $$Q(\alpha)=\sum_{i=1}^n\left[\log\Gamma(\sum_{j=1}^K\alpha_j)-\sum_{k=1}^K\log\Gamma(\alpha_k)+\sum_{k=1}^K(\alpha_k-1)\log(\mu_{ik})+\log(\frac{N!}{y_{i1}!...y_{iK}!})+\sum_{k=1}^K y_{ik}\log(\mu_{ik})\right].$$ $$\frac{dQ}{d\alpha_k}=\sum_{i=1}^n\left[\psi(\sum_{j=1}^K\alpha_j)-\psi(\alpha_k)+\psi(\nu_{ik})-\psi(\sum_{j=1}^K\nu_{ij})\right]=N\left[\psi(\sum_{j=1}^K\alpha_j)-\psi(\alpha_k)\right]+\sum_{i=1}^N\left[\psi(\nu_{ik})-\psi(\sum_{j=1}^K\nu_{ij})\right].$$ Setting $\frac{dQ}{d\alpha_k}=0$ would not give closed-form updating approach for parameter $\alpha$, so we need to use iterative method in the M-step.

The MLE is a convex optimization problem. An easy way to see this is making use of the property of the exponential family. Recall that the exponential family of random variable $x$ is define as $$p(x|\eta)=h(x)\exp\{\eta^T T(x) - A(\eta)\},$$ where $\eta$ is called the natural parameter, and $T(x)$ is the sufficient statistics and $A(\eta)$ is the log normalizer. Some common distributions which belong to this family are the Gaussian, Bernoulli and Multinormial distributions. The Dirichlet distribution here again takes this form by writing: \begin{align*} h(x)&=1\\ \eta &= \alpha-1\\ T(x) &= \log p\\ A(\eta) & = N\left(\sum_k \log \Gamma(\alpha_k)- \log \Gamma\left(\sum_k \alpha_k\right)\right) \end{align*} An important property for the exponential family is that $A(\eta)$ is the cumulant generating function for the sufficient statistics. It means that $A'(\eta)$ is the expectation, and $A''(\eta)$ is the variance. Note that variance is always non-negative, implying $A(\eta)$ is convex and thus the log-likelihood is a convex function in $\eta$. An alternative way is evaluate the hessian matrix explicitly and show it is positive semi-definite.

(b)¶

Let $\beta_k$ be the $k$-th column vector of the matrix $\beta$. Let $\alpha_{ik}=\exp(\beta_k^Tx_i)$.

E-step: $$p(\mu_i|y_i,x_i,\beta)\propto p(\mu_i|x_i,\beta)p(y_i|\mu_i),$$ implies the updating equation for the E-step as $$q(\mu_i)=\text{Dir}(\mu_i|(\nu_{i1},\nu_{i2},...,\nu_{iK})),$$ $$\nu_{ik}=y_{ik}+\exp(x_i^T\beta_k).$$

M-step: $$\begin{aligned}Q(\beta) &=\mathbb{E}_q\sum_i\left[\log(\frac{N!}{y_{i1}!...y_{iK}!})+\sum_k y_{ik}\log(\mu_{ik})+\log\Gamma(\sum_k e^{x_i^T\beta_k})-\sum_k\log\Gamma(e^{x_i^T\beta_k})+\sum_k(e^{x_i^T\beta_k}-1)\log(\mu_{ik})\right]\\ &=\sum_i\left[\log(\frac{N!}{y_{i1}!...y_{iK}!})+\log\Gamma(\sum_k e^{x_i^T\beta_k})-\sum_k\log\Gamma(e^{x_i^T\beta_k})+\sum_k(y_{ik}+e^{x_i^T\beta_k}-1)(\psi(\nu_{ik})-\psi(\sum_k\nu_{ik}))\right]. \end{aligned}$$ $$\frac{\partial Q}{\partial\beta_k}=\sum_i\left[\psi(\sum_k e^{x_i^T\beta_k})-\psi(e^{x_i^T\beta_k})+\psi(\nu_{ik})-\psi(\sum_k\nu_{ik})\right]e^{x_i^T\beta_k}x_i.$$ Setting $\frac{\partial Q}{\partial\beta_k}=0$ does not yield the closed-form solution, so we need to use iterative method in M-step. Here, we use Newton-Raphson method, the Hessian matrix is obtained as $$\frac{\partial^2 Q}{\partial\beta_k\partial\beta_k^T}=\sum_i\left[\psi'(\sum_ke^{x_i^T\beta_k})-\psi'(e^{x_i^T\beta_k})\right]e^{2x_i^T\beta_k}x_ix_i^T + \sum_i\left[\psi(\sum_ke^{x_i^T\beta_k})-\psi(e^{x_i^T\beta_k})+\psi(\nu_{ik})-\psi(\sum_k\nu_{ik})\right]e^{x_i^T\beta_k}x_ix_i^T.$$ The updating approach for M-step is given as $$\beta_k\leftarrow\beta_k-\left(\frac{\partial^2 Q}{\partial\beta_k\partial\beta_k^T}\right)^{-1}\frac{\partial Q}{\partial\beta_k}.$$

Lower bound: $$\mathcal{L}=\sum_i\left[\log(\frac{N!}{y_{i1}!...y_{iK}!})+\log\Gamma(\sum_k e^{x_i^T\beta_k})-\sum_k\log\Gamma(e^{x_i^T\beta_k})-\log\Gamma(\sum_k\nu_{ik})+\sum_k\log\Gamma(\nu_{ik})+\sum_k(y_{ik}+e^{x_i^T\beta_k}-\nu_{ik})(\psi(\nu_{ik})-\psi(\sum_k\nu_{ik}))\right].$$ At each iteration, after E-step and before M-step, $\mathcal{L}$ equals to the log-likelihood. So we check the monotonicity of the log-likelihood and convergence of EM algorithm by checking $\mathcal{L}$ at each iteration (after E-step and before M-step).

from numpy.linalg import inv
from scipy.special import gammaln
from scipy.special import polygamma
#digamma polygamma(0, x); trigamma polygamma(1, x)

# Pre-specify
XXX = inv(X.T @ X) @ X.T
n_iter = 500
tol = 1e-6
lbs = np.zeros(n_iter)

# Initialization
beta = XXX @ np.log(Y + (Y==0))

# EM algorithm
for iter in range(n_iter):
    # E-step
    eXb = np.exp(X @ beta)
    V = Y + eXb
    # Lower bound
    lb = np.sum(
        gammaln(np.sum(eXb, axis=1)) - np.sum(gammaln(eXb), axis=1) - \
        gammaln(np.sum(V, axis=1)) + np.sum(gammaln(V), axis=1)
    )
    lbs[iter] = lb
    # M-step
    seXb = np.sum(eXb, axis=1)
    sV = np.sum(V, axis=1)
    for k in range(K):
        tmp1 = (polygamma(0, seXb) - polygamma(0, eXb[:,k]) + polygamma(0, V[:,k]) - polygamma(0, sV)) * eXb[:,k]
        tmp2 = (polygamma(1, seXb) - polygamma(1, eXb[:,k])) * (eXb[:,k] ** 2)
        f = tmp1.reshape((1,n)) @ X
        H = ((tmp1+tmp2) * X.T) @ X
        beta[:,k] = beta[:,k] - (inv(H) @ f.T)[:,0]
